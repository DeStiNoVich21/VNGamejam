// ТЕСТ СЛОЁВ CANVAS-MAIN
// Проверяет: включение/выключение dialogue и characters
// во время активного диалога
// ═══════════════════════════════════════════════════

// --- БЛОК 1: Всё включено (базовый режим) ---
// Текст виден, спрайты видны
1 "Блок первый. Всё включено — текст и спрайты."

// Появляется Варг
CreateCharacter(Varg)
Varg.SetPosition(0.85 0.18)
Varg.Show()

Varg "Капитан. Вы меня видите и слышите?"
1 "Вижу и слышу. Продолжаем тест."

// --- БЛОК 2: Только текст, спрайты выключены ---
layer.SetActive(characters false)

1 "Блок второй. Спрайты выключены — только текст."
Varg "Вы меня не видите, но слышите голос."
1 "Странно разговаривать с пустотой."

// Возвращаем спрайты
layer.SetActive(characters true)
Varg "Я вернулся."

// --- БЛОК 3: Только спрайты, текст выключен ---
layer.SetActive(dialogue false)

// Диалог продолжается без текстового окна
Varg "Блок третий. Текст выключен — только спрайты и имя."
1 "Боксы диалога нет, но реплики идут."
Varg "Можно использовать для немых сцен."

// Возвращаем текст
layer.SetActive(dialogue true)
1 "Блок третий завершён. Текст вернулся."

// --- БЛОК 4: Всё выключено, диалог продолжается ---
layer.SetActive(dialogue false)
layer.SetActive(characters false)

// Пустой экран — только фон, диалог идёт невидимо
1 "Блок четвёртый. Пустой экран. Нажми пробел чтобы идти дальше."
Varg "Диалог активен, canvas пустой."
1 "Используй для переходов или внутреннего монолога без UI."

// Возвращаем всё
layer.SetActive(dialogue true)
layer.SetActive(characters true)

// --- БЛОК 5: Fade вместо резкого переключения ---
1 "Блок пятый. Плавный fade."

layer.FadeTo(characters 0 -spd 1.5)
Varg "Я плавно исчезаю..."

layer.FadeTo(characters 1 -spd 1.5)
Varg "И плавно возвращаюсь."

layer.FadeTo(dialogue 0 -spd 2.0)
1 "Текст уходит плавно."
layer.FadeTo(dialogue 1 -spd 2.0)
1 "Текст вернулся."

// --- ФИНАЛ ---
Varg "Тест завершён, Капитан."
Varg.Hide()
1 "Все слои работают. Идём дальше."
